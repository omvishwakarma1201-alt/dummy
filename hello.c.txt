#include <stdio.h>

int main() {
    char name[50];
    int age;
    char course[50];

    // Input
    printf("Enter Student Name: ");
    fgets(name, sizeof(name), stdin);

    printf("Enter Age: ");
    scanf("%d", &age);

    getchar(); // clear newline from buffer
    printf("Enter Course: ");
    fgets(course, sizeof(course), stdin);

    // Output (Profile Screen)
    printf("\n======= STUDENT PROFILE =======\n");
    printf("Name   : %s", name);
    printf("Age    : %d\n", age);
    printf("Course : %s", course);
    printf("================================\n");

    return 0;
}
